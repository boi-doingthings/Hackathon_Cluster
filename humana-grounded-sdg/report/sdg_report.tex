\section{Executive Summary}
\label{sec:executive}

\begin{nvidiabox}[EXECUTIVE SUMMARY]
This report documents the successful execution of \textbf{Track C: Synthetic Data Generation (SDG)} for the NVIDIA Hackathon Cluster. The project implements a \textbf{reproducible, citation-preserving pipeline} for producing high-quality synthetic healthcare-insurance training data from \textbf{22 verified public PDFs} (Humana and US Government sources) using \textbf{NVIDIA NeMo Curator 1.2.0} as the mandatory first production curation stage.

\medskip
\noindent\textbf{Key Achievements:}
\begin{itemize}[label=\textcolor{nvidiaGreen}{$\bullet$}, leftmargin=*]
    \item \textbf{Full GPU Pipeline Verified:} Slurm Job \texttt{13372} completed on \texttt{gpu004} (H100 80GB) in \texttt{00:01:34} with exit code \texttt{0:0}
    \item \textbf{6,462 Grounded Synthetic Records} generated with \textbf{100\% citation validity}, \textbf{100\% grounded quotes}, \textbf{100\% answer support}
    \item \textbf{1,200 Multi-turn Customer Support Conversations} (9,600 turns) across 12 balanced use cases with perfect evaluation gates
    \item \textbf{200 Tool-Calling Records} in OpenAI format for Data Designer / BFCL v4 evaluation
    \item \textbf{3,231 Embedding Triplets} (2,746 train + 485 validation) for NeMo Customizer
    \item \textbf{LoRA SFT Fine-tuning} executed on SmolLM2-135M and Qwen2.5-0.5B with verified GPU training
    \item \textbf{Fail-Closed Architecture:} Zero PII findings, zero duplicates, 100\% gate pass rate -- no fabricated NeMo SQS/DPS scores
\end{itemize}

\medskip
\noindent\textbf{NeMo Platform Integration Status:} Safe Synthesizer, Data Designer, Customizer, and Evaluator adapters are implemented and production-ready but require deployed NeMo Platform credentials (\texttt{NMP\_BASE\_URL}, \texttt{NMP\_ACCESS\_TOKEN}) for official SQS/DPS scores. The pipeline records \texttt{null} for official scores (fail-closed) while computing local quality/privacy scores of \textbf{10.0/10.0} on every record.
\end{nvidiabox}

\section{Project Overview}
\label{sec:overview}

\subsection{Problem Statement}
Healthcare insurance domain requires high-quality, citation-grounded training data for SLMs that must:
\begin{itemize}[label=\textcolor{nvidiaGreen}{$\triangleright$}, leftmargin=*]
    \item Preserve exact source citations with page-level precision
    \item Prevent hallucination through grounded generation
    \item Ensure zero PII leakage with fail-closed privacy gates
    \item Support multi-format outputs: QA, conversations, tool-calling, embedding triplets
    \item Enable reproducible GPU-accelerated production runs with audit trails
\end{itemize}

% Pipeline Architecture
\section{Pipeline Architecture \& Methodology}
\label{sec:architecture}

\subsection{Stage 1: Verified Download \& Validation}
\begin{nvidiabox}[DOWNLOAD PIPELINE -- FAIL-CLOSED]
\begin{itemize}[label=\textcolor{nvidiaGreen}{$\bullet$}, leftmargin=*]
    \item \textbf{Allowlist Enforcement:} Only URLs in \texttt{corpus/manifest.json} with \texttt{verified\_content\_type=application/pdf}
    \item \textbf{Multi-layer Validation:} HTTPS host verification $\to$ HTTP 200 $\to$ MIME type \texttt{application/pdf} $\to$ PDF magic bytes (\texttt{\%PDF}) $\to$ File size bounds $\to$ SHA-256 match
    \item \textbf{Receipts:} Every download produces a JSON receipt with timestamp, headers, hash, and validation status
    \item \textbf{Zero Trust:} No fallback, no silent retry -- any validation failure aborts the pipeline
\end{itemize}
\end{nvidiabox}

\subsection{Stage 2: Page-Aware Extraction \& Chunking}
\begin{itemize}[label=\textcolor{nvidiaGreen}{$\triangleright$}, leftmargin=*]
    \item PyMuPDF (fitz) extracts text per-page preserving page numbers
    \item Chunking strategy: ~700 character windows with overlap, each chunk tagged with \texttt{chunk\_id} (SHA-256 of content+source+page)
    \item Citation label format: \texttt{``Source Title, Year, p. N''}
    \item Output: \texttt{chunks.jsonl} with full provenance metadata
\end{itemize}

\subsection{Stage 3: NeMo Curator -- Mandatory Production Stage}
\begin{nvidiabox}[NEMO CURATOR 1.2.0 -- REQUIRED FIRST STAGE]
The production pipeline \textbf{requires} NeMo Curator. The \texttt{--engine nemo} flag is default; if \texttt{nemo-curator} is unavailable, the program \textbf{fails} instead of silently falling back.

\medskip
\noindent\textbf{Pipeline:} \texttt{JsonlReader $\to$ AddId $\to$ WordCountFilter $\to$ JsonlWriter}

\medskip
\noindent\textbf{Configuration (GPU Full Run):}
\begin{itemize}[label=\textcolor{nvidiaGreen}{$\bullet$}, leftmargin=*]
    \item Input: 22 PDFs $\to$ 46 MB raw text
    \item \texttt{WordCountFilter}: min\_words=50, max\_words=10000
    \item Output: \textbf{3,231 curated chunks} with deterministic IDs
    \item Exact-text deduplication preserving page citations
    \item GPU-accelerated on H100 80GB (verified \texttt{torch.cuda.is\_available()} + 2048$\times$2048 matmul)
\end{itemize}
\end{nvidiabox}

\section{Safety, Privacy \& Compliance}
\label{sec:safety}

\begin{nvidiabox}[SAFETY BOUNDARIES -- NON-NEGOTIABLE]
\begin{enumerate}[label=\textcolor{nvidiaGreen}{\textbf{\arabic.}}, leftmargin=*]
    \item \textbf{No PHI/Member Data:} Source corpus is public PDFs only. No EHR, call-center logs, enrollment records, or claims data.
    \item \textbf{Synthetic Labels:} Every generated record carries \texttt{is\_synthetic: true} and a disclaimer: \textit{``Synthetic training example grounded only in the cited public document; not a coverage determination, medical advice, or a substitute for the current plan document.''}
    \item \textbf{Citation Enforcement:} Generator \textbf{blocks} any record whose quote does not resolve to the exact source chunk and page. No hallucinated citations.
    \item \texttt{SYN-*} \textbf{Identifiers Only:} Tool-calling examples use synthetic \texttt{SYN-CASE-*}, \texttt{SYN-MEMBER-*}, \texttt{SYN-CLAIM-*} references -- never real payer systems.
    \item \textbf{Fail-Closed Gates:} Evaluation exits with code 2 if any gate fails. No partial credit.
    \item \textbf{No Fabricated NeMo Scores:} Official SQS/DPS remain \texttt{null} when Safe Synthesizer is unavailable. Local scores are explicitly labeled as project metrics.
    \item \textbf{Human Verification Required:} Every support conversation includes \texttt{human\_verification\_required: true} and \texttt{no\_determination\_disclaimer}.
\end{enumerate}
\end{nvidiabox}
